--[[
    MUBASHIR HUB - STABLE VERSION
    Fixed Execution Logic & Auto-Loading Features
]]

local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
-- Ensure character and GUI are ready before starting
local playerGui = player:WaitForChild("PlayerGui", 10)
if not playerGui then return end

-- // CONFIGURATION //
local currentKeybind = Enum.KeyCode.RightAlt
local isVisible = true
local speedConnection = nil
local lagEnabled = false

-- // UI SETUP //
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "MubashirHubFixed"
screenGui.ResetOnSpawn = false
screenGui.Parent = playerGui

local mainFrame = Instance.new("Frame")
mainFrame.Size = UDim2.new(0, 230, 0, 400)
mainFrame.Position = UDim2.new(0.5, -115, 0.5, -200)
mainFrame.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
mainFrame.BorderSizePixel = 0
mainFrame.Active = true
mainFrame.Parent = screenGui

local mainCorner = Instance.new("UICorner", mainFrame)
mainCorner.CornerRadius = UDim.new(0, 10)

local header = Instance.new("Frame")
header.Size = UDim2.new(1, 0, 0, 40)
header.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
header.BorderSizePixel = 0
header.Parent = mainFrame

local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, 0, 1, 0)
title.BackgroundTransparency = 1
title.Text = "MUBASHIR HUB FIXED"
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.Font = Enum.Font.GothamBold
title.TextSize = 14
title.Parent = header

local scroll = Instance.new("ScrollingFrame")
scroll.Size = UDim2.new(1, -10, 1, -50)
scroll.Position = UDim2.new(0, 5, 0, 45)
scroll.BackgroundTransparency = 1
scroll.CanvasSize = UDim2.new(0, 0, 0, 600)
scroll.ScrollBarThickness = 2
scroll.Parent = mainFrame

local layout = Instance.new("UIListLayout", scroll)
layout.Padding = UDim.new(0, 5)

-- // BUTTON BUILDER //
local function createButton(name, color, callback)
    local btn = Instance.new("TextButton")
    btn.Size = UDim2.new(1, -10, 0, 35)
    btn.BackgroundColor3 = color or Color3.fromRGB(25, 25, 25)
    btn.Text = name
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = Enum.Font.Gotham
    btn.TextSize = 12
    btn.Parent = scroll
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 6)
    
    btn.MouseButton1Click:Connect(function()
        pcall(function() callback(btn) end)
    end)
    return btn
end

-- // FEATURES //

-- 1. AUTO BLOCK (Automatic loadstring execution)
createButton("Auto Block (GOATED)", Color3.fromRGB(40, 80, 40), function(btn)
    btn.Text = "Loading..."
    task.spawn(function()
        local success, err = pcall(function()
            loadstring(game:HttpGet("https://raw.githubusercontent.com/Tokinu/TP-BLOCK/refs/heads/main/GOATED"))()
        end)
        if success then
            btn.Text = "GOATED Active"
        else
            btn.Text = "Failed to Load"
            warn(err)
        end
    end)
end)

-- 2. TIGY SPEED (Stabilized)
createButton("Tigy Speed Boost", nil, function(btn)
    if speedConnection then
        speedConnection:Disconnect()
        speedConnection = nil
        btn.Text = "Tigy Speed (OFF)"
        btn.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
    else
        btn.Text = "Tigy Speed (ON)"
        btn.BackgroundColor3 = Color3.fromRGB(0, 120, 200)
        speedConnection = RunService.Heartbeat:Connect(function()
            local char = player.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            if hrp and hum and hum.MoveDirection.Magnitude > 0 then
                hrp.Velocity = Vector3.new(hum.MoveDirection.X * 55, hrp.Velocity.Y, hum.MoveDirection.Z * 55)
            end
        end)
    end
end)

-- 3. BK LAGGER
createButton("BK Lagger", Color3.fromRGB(80, 20, 20), function(btn)
    lagEnabled = not lagEnabled
    btn.Text = lagEnabled and "LAGGING..." or "BK Lagger"
    task.spawn(function()
        while lagEnabled do
            local net = ReplicatedStorage:FindFirstChild("Packages") and ReplicatedStorage.Packages:FindFirstChild("Net")
            local useItem = net and net:FindFirstChild("RE/UseItem")
            if useItem then
                for i = 1, 20 do useItem:FireServer() end
            end
            task.wait(0.1)
        end
    end)
end)

-- 4. UTILITIES
createButton("Fly To Base", nil, function()
    local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    if hrp then hrp.CFrame = CFrame.new(-331.9, -4.6, 19.1) end
end)

createButton("Reset Character", Color3.fromRGB(100, 50, 0), function()
    if player.Character then player.Character:BreakJoints() end
end)

-- // KEYBIND SYSTEM //
local bindBtn = createButton("Bind: RightAlt", Color3.fromRGB(40, 40, 40), function() end)
local binding = false

bindBtn.MouseButton1Click:Connect(function()
    binding = true
    bindBtn.Text = "..."
end)

UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if binding and input.UserInputType == Enum.UserInputType.Keyboard then
        currentKeybind = input.KeyCode
        bindBtn.Text = "Bind: " .. input.KeyCode.Name
        binding = false
    elseif input.KeyCode == currentKeybind then
        isVisible = not isVisible
        mainFrame.Visible = isVisible
    end
end)

-- // DRAGGING SYSTEM //
local dragging, dragStart, startPos
header.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
end)

print("Mubashir Hub Fixed & Loaded.")
